SELECT
product['product_name'],
product_product_family,
line_item_operation,
pricing_unit,
product['description'],
product['usage_family'],
product['pricingplan'],
FROM
${tableName}
WHERE
line_item_line_item_type = 'Usage'
GROUP BY
product['product_name'],
product_product_family,
line_item_operation,
pricing_unit,
product['description'],
product['usage_family'],
product['pricingplan'],
ORDER BY
product['product_name'],
product_product_family,
line_item_operation,
pricing_unit,
product['description'],
product['usage_family'],
product['pricingplan'],
